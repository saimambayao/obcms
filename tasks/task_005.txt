Model example:
```python
from django.db import models

class Stakeholder(models.Model):
    STAKEHOLDER_TYPES = [
        ('community_leader', 'Community Leader'),
        ('ulama', 'Ulama'),
        ('imam', 'Imam'),
        ('arabic_teacher', 'ALIVE/Arabic Teacher'),
        ('youth_leader', 'Youth Leader'),
        ('women_leader', 'Women Leader'),
        ('other', 'Other')
    ]
    
    full_name = models.CharField(max_length=255)
    stakeholder_type = models.CharField(max_length=50, choices=STAKEHOLDER_TYPES)
    community = models.ForeignKey('OBCCommunity', on_delete=models.CASCADE, related_name='stakeholders')
    position = models.CharField(max_length=255, blank=True)
    contact_number = models.CharField(max_length=20, blank=True)
    email = models.EmailField(blank=True)
    address = models.TextField(blank=True)
    influence_level = models.CharField(max_length=20, choices=[
        ('high', 'High'),
        ('medium', 'Medium'),
        ('low', 'Low')
    ], default='medium')
    notes = models.TextField(blank=True)
    is_active = models.BooleanField(default=True)
    since_year = models.PositiveIntegerField(null=True, blank=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
```
