```python
from django.db import models
from django.conf import settings

class MANAAssessment(models.Model):
    STATUS_CHOICES = [
        ('planned', 'Planned'),
        ('in_progress', 'In Progress'),
        ('completed', 'Completed'),
        ('analyzed', 'Analyzed'),
        ('reported', 'Reported')
    ]
    
    METHODOLOGY_CHOICES = [
        ('survey', 'Survey'),
        ('interview', 'Interviews'),
        ('focus_group', 'Focus Group Discussion'),
        ('observation', 'Direct Observation'),
        ('mixed', 'Mixed Methods'),
        ('other', 'Other')
    ]
    
    title = models.CharField(max_length=255)
    community = models.ForeignKey('OBCCommunity', on_delete=models.CASCADE, related_name='assessments')
    description = models.TextField()
    methodology = models.CharField(max_length=20, choices=METHODOLOGY_CHOICES)
    start_date = models.DateField()
    end_date = models.DateField(null=True, blank=True)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='planned')
    lead_assessor = models.ForeignKey(settings.AUTH_USER_MODEL, on_delete=models.SET_NULL, null=True, related_name='led_assessments')
    team_members = models.ManyToManyField(settings.AUTH_USER_MODEL, related_name='assessment_teams', blank=True)
    objectives = models.TextField()
    notes = models.TextField(blank=True)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
```
